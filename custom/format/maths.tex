\usepackage{mathtools}
\usepackage{cleveref}

%
% Constraints
%
\newcounter{constraint}
\crefalias{constraint}{const}
\creflabelformat{const}{#2C#1#3}
\crefname{const}{constraint}{constraints}

\newcommand{\cstlabel}[1]{\label[const]{#1}} % chktex 10 % chktex 9

\newtagform{Const}{(C}{)}

\makeatletter
\NewDocumentEnvironment{Constraint}{}{%
  \usetagform{Const}%
  \let\c@equation\c@constraint\def\theequation{\theconstraint}%
}{}
\makeatother

%
% Objective functions
%
\newcounter{objective}

\crefalias{objective}{obj}
\creflabelformat{obj}{#2#1#3}
\crefname{obj}{objective}{objectives}

\newcommand{\objlabel}[1]{\label[obj]{#1}} % chktex 10 % chktex 9

\newtagform{Obj}{(O}{)}

\makeatletter
\NewDocumentEnvironment{Objective}{}{%
  \usetagform{Obj}%
  \let\c@equation\c@objective\def\theequation{\theobjective}%
}{}
\makeatother
