%
% Usage: \Set*{... \given ...} or \Set[\big]{...}
%
\providecommand\given{} % just to make sure it exists
\newcommand\SetSymbol[1][]{%
  \nonscript{}\:#1\vert%
  \allowbreak%
  \nonscript{}\:
  \mathopen{}
}
\DeclarePairedDelimiterX{\Set}[1]{\{}{\}}{%
  \renewcommand\given{\SetSymbol[\delimsize]}
  #1
}
