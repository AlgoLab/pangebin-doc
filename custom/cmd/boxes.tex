\newcommand{\boxfactory}[3]{
    \newtcolorbox{#1}{
      empty,
      breakable,
      % Title ---
      boxed title style={
        empty,
        left*=0pt,
      },
      title=#2,
      coltitle=#3,
      fonttitle=\sffamily\bfseries,
      % Inner text ---
      parbox=false,
      coltext=drawio_grey,
      fontupper=\sffamily,
      % Border line ---
      borderline west={2\heavyrulewidth}{0pt}{#3},
    }
}
